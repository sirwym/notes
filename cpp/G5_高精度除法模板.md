```cpp
#include <bits/stdc++.h>
using namespace std;

// 高精度 / 低精度，返回商，余数通过引用返回
vector<int> div(vector<int> &A, int b, int &r) {
    vector<int> C;
    r = 0;
    for (int i = A.size() - 1; i >= 0; i--) {  // 从高位开始
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }
    reverse(C.begin(), C.end());  // 翻转回低位在前
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

int main() {
    string a; int b;
    cin >> a >> b;
    vector<int> A;
    for (int i = a.size()-1; i >= 0; i--) A.push_back(a[i]-'0');

    int remainder;
    auto C = div(A, b, remainder);

    for (int i = C.size()-1; i >= 0; i--) cout << C[i];
    cout << endl << remainder << endl;
    return 0;
}
```

## 📖 要点说明

- 除法从高位开始，与加减乘不同
- 余数 `r = r*10 + 当前位`，商 = `r/b`
- 结果需 reverse 回低位在前的统一格式

### ⚠️ 常见错误

- 从低位开始除（应该从高位）
- 忘 reverse 结果
- 不去前导零
